const https = require('https');
const fs = require('fs');
const crypto = require('crypto');

// Read ASC credentials from workflow
const workflow = fs.readFileSync(
  '.github/workflows/ios-testflight.yml', 'utf8');

// Extract key ID and issuer
const keyIdMatch = workflow.match(
  /ASC_API_KEY_ID['":\s]+\$\{\{[^}]+\}\}/);
const issuerMatch = workflow.match(
  /ASC_API_KEY_ISSUER_ID['":\s]+\$\{\{[^}]+\}\}/);

console.log('Key ID pattern:', keyIdMatch?.[0]);
console.log('Issuer pattern:', issuerMatch?.[0]);

// Check if we have the actual values in env
const keyId = process.env.ASC_API_KEY_ID;
const issuerId = process.env.ASC_API_KEY_ISSUER_ID;
const keyP8 = process.env.ASC_KEY_P8;

if (!keyId || !issuerId || !keyP8) {
  console.log('ASC API credentials not in environment');
  console.log('keyId:', keyId ? 'SET' : 'NOT SET');
  console.log('issuerId:', issuerId ? 'SET' : 'NOT SET');
  console.log('keyP8:', keyP8 ? 'SET' : 'NOT SET');
  process.exit(1);
}

// Create JWT
function createJWT(keyId, issuerId, privateKeyPem) {
  const header = Buffer.from(JSON.stringify(
    { alg: 'ES256', kid: keyId, typ: 'JWT' }
  )).toString('base64url');

  const now = Math.floor(Date.now() / 1000);
  const payload = Buffer.from(JSON.stringify({
    iss: issuerId,
    iat: now,
    exp: now + 1200,
    aud: 'appstoreconnect-v1'
  })).toString('base64url');

  const sign = crypto.createSign('SHA256');
  sign.update(`${header}.${payload}`);
  const signature = sign.sign(privateKeyPem, 'base64url');

  return `${header}.${payload}.${signature}`;
}

const token = createJWT(keyId, issuerId, keyP8);

// Query app info
function queryASC(path) {
  return new Promise((resolve, reject) => {
    const options = {
      hostname: 'api.appstoreconnect.apple.com',
      path: `/v1/${path}`,
      headers: { 'Authorization': `Bearer ${token}` }
    };
    https.get(options, (res) => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => resolve(JSON.parse(data)));
    }).on('error', reject);
  });
}

async function main() {
  // Get app info
  const app = await queryASC('apps/6760251165');
  const bundleId = app.data?.attributes?.bundleId;
  const appName = app.data?.attributes?.name;
  console.log('App name:', appName);
  console.log('Bundle ID in ASC:', bundleId);

  // Get IAP products
  const iaps = await queryASC(
    'apps/6760251165/inAppPurchasesV2?limit=20');
  console.log('\nIAP Products:');
  iaps.data?.forEach(iap => {
    console.log('-', iap.attributes?.productId,
      '|', iap.attributes?.state);
  });
}

main().catch(console.error);

```bash
#!/bin/bash
# ASC Status Check — queries App Store Connect API
# Usage: ./scripts/asc-status.sh
# Requires: ASC_API_KEY_ID, ASC_API_KEY_ISSUER_ID,
#           ASC_KEY_P8 in environment

set -e

APP_ID="6760251165"

if [ -z "$ASC_API_KEY_ID" ]; then
  echo "ERROR: ASC_API_KEY_ID not set"
  echo "These are GitHub Actions secrets."
  echo "Run this from CI or set manually."
  exit 1
fi

echo "Querying ASC for app $APP_ID..."

# Generate JWT
python3 - << 'PYEOF'
import jwt, time, os, urllib.request, json

key_id = os.environ['ASC_API_KEY_ID']
issuer_id = os.environ['ASC_API_KEY_ISSUER_ID']
private_key = os.environ['ASC_KEY_P8']

payload = {
    'iss': issuer_id,
    'iat': int(time.time()),
    'exp': int(time.time()) + 1200,
    'aud': 'appstoreconnect-v1'
}
token = jwt.encode(payload, private_key,
    algorithm='ES256',
    headers={'kid': key_id})

headers = {'Authorization': f'Bearer {token}'}

# Get app info
req = urllib.request.Request(
    f'https://api.appstoreconnect.apple.com/v1/apps/{os.environ.get("APP_ID", "6760251165")}',
    headers=headers)
with urllib.request.urlopen(req) as r:
    data = json.loads(r.read())
    attrs = data['data']['attributes']
    print(f"App: {attrs['name']}")
    print(f"Bundle ID: {attrs['bundleId']}")
    print(f"SKU: {attrs['sku']}")

# Get IAP
req2 = urllib.request.Request(
    f'https://api.appstoreconnect.apple.com/v1/apps/{os.environ.get("APP_ID", "6760251165")}/inAppPurchasesV2?limit=20',
    headers=headers)
with urllib.request.urlopen(req2) as r:
    data = json.loads(r.read())
    print(f"\nIAP Products ({len(data['data'])} found):")
    for iap in data['data']:
        a = iap['attributes']
        print(f"  - {a['productId']} | {a['name']} | {a['state']}")
PYEOF
```
