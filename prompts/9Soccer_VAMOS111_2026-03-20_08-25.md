  app.get_beta_groups.each do |group|

    unless group.public_link_enabled || group.is_internal_group
      puts "  Enabling public link..."
      group.public_link_enabled = true
      group.save!
      puts "  Public link: #{group.public_link}"
    end
  end
rescue => e
  puts "Error: #{e.message}"
end
RUBY

### B2. If spaceship fails, try via ASC API directly:
```bash
# Check if ASC API key is available in environment
grep -r "ASC_API_KEY_ID\|ASC_ISSUER" \
  /c/Projects/90soccer/.github/workflows/ios-testflight.yml | head -5

# Try with JWT auth
ruby << 'RUBY'
require 'jwt'
require 'net/http'
require 'json'
require 'openssl'

api_key_id = ENV['ASC_API_KEY_ID']
issuer_id = ENV['ASC_API_KEY_ISSUER_ID']
key_p8 = ENV['ASC_KEY_P8']

if api_key_id && issuer_id && key_p8
  private_key = OpenSSL::PKey::EC.new(key_p8)
  payload = {
    iss: issuer_id,
    iat: Time.now.to_i,
    exp: Time.now.to_i + 1200,
    aud: 'appstoreconnect-v1'
  }
  token = JWT.encode(payload, private_key, 'ES256',
    { kid: api_key_id })

  # Get beta groups
  uri = URI('https://api.appstoreconnect.apple.com/v1/betaGroups?filter[app]=6760251165')
  req = Net::HTTP::Get.new(uri)
  req['Authorization'] = "Bearer #{token}"
  req['Content-Type'] = 'application/json'

  res = Net::HTTP.start(uri.hostname, uri.port, use_ssl: true) { |h| h.request(req) }
  data = JSON.parse(res.body)

  data['data']&.each do |group|
    puts "Group: #{group.dig('attributes', 'name')}"
    puts "  Public link: #{group.dig('attributes', 'publicLink') || 'not enabled'}"
    puts "  Public link enabled: #{group.dig('attributes', 'publicLinkEnabled')}"
    puts "  ID: #{group['id']}"
  end
else
  puts "ASC API keys not in environment"
  puts "api_key_id: #{api_key_id ? 'set' : 'NOT SET'}"
  puts "issuer_id: #{issuer_id ? 'set' : 'NOT SET'}"
end
RUBY
```
