One-tap WhatsApp challenge:
```typescript
// src/lib/friend-challenge.ts
export function createChallenge(userId: string, challengeId: number, score: number) {
  const code = btoa(`${userId}-${challengeId}-${score}`).slice(0, 8)
  return {
    url: `https://9soccer.vercel.app/?challenge=${code}`,
    whatsapp: `https://wa.me/?text=${encodeURIComponent(
      `⚽ קיבלתי ${score}/5 על "${getChallengeTitle(challengeId)}" ב-9Soccer!\n` +
      `🏆 האם אתה יכול לנצח אותי?\n` +
      `▶️ https://9soccer.vercel.app/?challenge=${code}`
    )}`
  }
}
```
