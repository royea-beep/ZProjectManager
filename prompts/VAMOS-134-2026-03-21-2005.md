### Add Weekend Tournament system

**Logic:**
- Friday 20:00 IST → Saturday 23:59 IST = Weekend Tournament window
- Top 3 players by total BR score over weekend win prize pool
- Prize: 1st = 500 coins + "Weekend Champion" badge, 2nd = 250, 3rd = 100
- Leaderboard resets every Monday 00:00 IST

**Implementation:**
```typescript
// src/lib/weekend-tournament.ts
export function isWeekendTournament(): boolean {
  const now = new Date()
  // Convert to IST (UTC+3)
  const ist = new Date(now.getTime() + 3 * 60 * 60 * 1000)
  const day = ist.getDay() // 5=Friday, 6=Saturday
  const hour = ist.getHours()
  return (day === 5 && hour >= 20) || day === 6
}

export function getWeekendPrize(rank: 1 | 2 | 3): { coins: number; badge: string } {
  const prizes = {
    1: { coins: 500, badge: '🏆 Weekend Champion' },
    2: { coins: 250, badge: '🥈 Weekend Runner-up' },
    3: { coins: 100, badge: '🥉 Weekend Finalist' },
  }
  return prizes[rank]
}
```
