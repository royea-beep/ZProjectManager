### A3. If screenshots are wrong size, resize them:
The iOS Simulator on macos-latest outputs at device
native resolution. Use Node.js sharp or Python Pillow
to resize to exactly 1290×2796 (6.9" required).

```bash
# Install sharp if needed
cd /c/Projects/90soccer
npm install --save-dev sharp 2>/dev/null

# Resize all screenshots
node -e "
const sharp = require('sharp');
const fs = require('fs');
const path = require('path');

const INPUT_DIR = './screenshots/app-store-screenshots-build-1';
const OUTPUT_DIR = './screenshots/asc-ready';

fs.mkdirSync(OUTPUT_DIR, { recursive: true });

const files = fs.readdirSync(INPUT_DIR)
  .filter(f => f.endsWith('.png'));

Promise.all(files.map(async (file) => {
  const input = path.join(INPUT_DIR, file);
  const output = path.join(OUTPUT_DIR, file);
  await sharp(input)
    .resize(1290, 2796, { fit: 'contain', background: '#0B0E18' })
    .toFile(output);
  console.log('Resized:', file, '→ 1290x2796');
})).then(() => console.log('Done! Files in:', OUTPUT_DIR));
"
```
