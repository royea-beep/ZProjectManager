### Common iPhone video issues:

**Issue 1 — CORS / format not supported:**
iPhone Safari supports: MP4 (H.264), HLS (.m3u8)
iPhone Safari does NOT support: WebM, VP9, AV1

```tsx
// Fix: ensure video source is MP4 H.264
<video
  src={clip.url}
  playsInline  // CRITICAL for iPhone — without this, goes fullscreen
  muted        // Required for autoplay on iOS
  autoPlay
  preload="auto"
  controls={false}
/>
```

**Issue 2 — Missing `playsInline`:**
Without `playsInline`, iPhone opens video in fullscreen player instead of inline.

**Issue 3 — Autoplay blocked:**
iOS requires user gesture OR muted video for autoplay.
```tsx
// Ensure: muted + playsInline + autoPlay all present
// Add: onCanPlay handler to start play
videoRef.current.play().catch(() => {
  // Show play button as fallback
  setShowPlayButton(true)
})
```

## PART 1.5 — GEM: YouTube → Google Drive 9-Second Clip Pipeline

Create `scripts/gem-youtube-to-drive.py` — a reusable tool for ALL projects.

### What it does:
1. Takes a YouTube URL + start_time (seconds) + clip_id
2. Downloads the video using yt-dlp (best quality MP4 H.264)
3. Cuts exactly 9 seconds starting from start_time using ffmpeg
4. Uploads the 9-second clip to Google Drive folder "9soccer-clips"
5. Returns the public Google Drive direct link
6. Deletes the original large file from disk
7. Updates Supabase: sets clip.video_url to the new Drive direct link

### Dependencies:
```bash
pip install yt-dlp google-auth google-auth-oauthlib google-api-python-client --break-system-packages
# ffmpeg must be installed: winget install ffmpeg
```

### Script structure:
```python
# scripts/gem-youtube-to-drive.py
import os, subprocess, sys
import yt_dlp
from googleapiclient.discovery import build
from googleapiclient.http import MediaFileUpload
from google.oauth2.service_account import Credentials

DRIVE_FOLDER_NAME = "9soccer-clips"
GDRIVE_CREDS = "scripts/gdrive-service-account.json"  # service account key

def download_youtube(url: str, output_path: str):
    """Download best MP4 H.264 from YouTube"""
    ydl_opts = {
        'format': 'bestvideo[ext=mp4][vcodec^=avc]+bestaudio[ext=m4a]/best[ext=mp4]',
        'outtmpl': output_path,
        'quiet': True,
    }
    with yt_dlp.YoutubeDL(ydl_opts) as ydl:
        ydl.download([url])

def cut_9_seconds(input_path: str, output_path: str, start_time: float):
    """Cut exactly 9 seconds using ffmpeg"""
    cmd = [
        'ffmpeg', '-y',
        '-ss', str(start_time),
        '-i', input_path,
        '-t', '9',
        '-c:v', 'libx264',  # H.264 — works on all devices including iPhone
        '-c:a', 'aac',
        '-movflags', '+faststart',  # web-optimized MP4
        '-vf', 'scale=720:-2',  # 720p max
        output_path
    ]
    subprocess.run(cmd, check=True, capture_output=True)

def upload_to_drive(file_path: str, clip_id: int) -> str:
    """Upload to Google Drive, return direct download URL"""
    creds = Credentials.from_service_account_file(GDRIVE_CREDS, 
        scopes=['https://www.googleapis.com/auth/drive'])
    service = build('drive', 'v3', credentials=creds)
    
    # Find or create folder
    folder_id = get_or_create_folder(service, DRIVE_FOLDER_NAME)
    
    file_metadata = {
        'name': f'clip_{clip_id:04d}_9s.mp4',
        'parents': [folder_id]
    }
    media = MediaFileUpload(file_path, mimetype='video/mp4', resumable=True)
    file = service.files().create(body=file_metadata, media_body=media, fields='id').execute()
    
    # Make public
    service.permissions().create(
        fileId=file['id'],
        body={'role': 'reader', 'type': 'anyone'}
    ).execute()
    
    # Return direct link (works on iPhone Safari)
    return f"https://drive.google.com/uc?export=download&id={file['id']}"

def process_clip(clip_id: int, youtube_url: str, start_time: float):
    """Full pipeline: YouTube → 9s clip → Drive → cleanup"""
    tmp_full = f"/tmp/clip_{clip_id}_full.mp4"
    tmp_clip = f"/tmp/clip_{clip_id}_9s.mp4"
    
    print(f"[{clip_id}] Downloading from YouTube...")
    download_youtube(youtube_url, tmp_full)
    
    print(f"[{clip_id}] Cutting 9 seconds from {start_time}s...")
    cut_9_seconds(tmp_full, tmp_clip, start_time)
    
    print(f"[{clip_id}] Uploading to Google Drive...")
    drive_url = upload_to_drive(tmp_clip, clip_id)
    
    print(f"[{clip_id}] Cleaning up...")
    os.remove(tmp_full)
    os.remove(tmp_clip)
    
    print(f"[{clip_id}] Done! URL: {drive_url}")
    return drive_url

# Example usage:
if __name__ == "__main__":
    # Test with one clip
    url = process_clip(
        clip_id=1,
        youtube_url="https://www.youtube.com/watch?v=EXAMPLE",
        start_time=45.0  # start at 45 seconds
    )
    print(url)
```
