## TASK A — iOS Simulator Screenshots via GitHub Actions
Agent role: Senior iOS CI Engineer who has built screenshot pipelines
for 50+ App Store submissions.

### A1. Create the workflow file:
Path: .github/workflows/screenshots.yml

```yaml
name: App Store Screenshots

on:
  workflow_dispatch:

jobs:
  screenshots:
    runs-on: macos-latest
    timeout-minutes: 60

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build static export
        run: npm run build

      - name: Install serve
        run: npm install -g serve

      - name: Start web server
        run: |
          serve out -p 3000 &
          echo "SERVER_PID=$!" >> $GITHUB_ENV
          sleep 3

      - name: Verify server is running
        run: |
          for i in {1..20}; do
            curl -sf http://localhost:3000 > /dev/null && echo "Server ready" && break
            echo "Waiting... ($i)"
            sleep 3
          done

      - name: List available simulators
        run: |
          xcrun simctl list devices available | grep iPhone | head -20

      - name: Boot best available iPhone simulator
        run: |
          # Try iPhone 16 Pro Max first, fall back to 15 Pro Max
          UDID=$(xcrun simctl list devices available | \
            grep -E "iPhone 1[56] Pro Max" | \
            grep -v "unavailable" | \
            head -1 | \
            grep -oE '[A-F0-9]{8}-[A-F0-9]{4}-[A-F0-9]{4}-[A-F0-9]{4}-[A-F0-9]{12}')

          if [ -z "$UDID" ]; then
            UDID=$(xcrun simctl list devices available | \
              grep "iPhone" | \
              grep -v "unavailable" | \
              head -1 | \
              grep -oE '[A-F0-9]{8}-[A-F0-9]{4}-[A-F0-9]{4}-[A-F0-9]{4}-[A-F0-9]{12}')
          fi

          echo "Using simulator: $UDID"
          echo "UDID=$UDID" >> $GITHUB_ENV
          xcrun simctl boot "$UDID"
          sleep 15
          xcrun simctl bootstatus "$UDID"

      - name: Open Safari and take screenshots
        run: |
          mkdir -p screenshots

          take_screenshot() {
            local url="$1"
            local filename="$2"
            local wait="${3:-4}"

            xcrun simctl openurl "$UDID" "http://localhost:3000${url}"
            sleep "$wait"
            xcrun simctl io "$UDID" screenshot "screenshots/${filename}.png"
            echo "✅ $filename"
          }

          # 1. Home Screen
          take_screenshot "/" "01_homescreen" 5

          # 2. Modes Hub
          take_screenshot "/modes" "02_modes_hub" 4

          # 3. Leaderboard
          take_screenshot "/leaderboard" "03_leaderboard" 4

          # 4. Profile
          take_screenshot "/profile" "04_profile" 4

          # 5. Pro Upgrade
          take_screenshot "/pro/upgrade" "05_pro_upgrade" 3

          # 6. Daily Streak
          take_screenshot "/daily-streak" "06_daily_streak" 3

          echo ""
          echo "Screenshots captured:"
          ls -la screenshots/*.png

      - name: Upload screenshots
        uses: actions/upload-artifact@v4
        with:
          name: app-store-screenshots-build-${{ github.run_number }}
          path: screenshots/
          retention-days: 90

      - name: Summary
        run: |
          echo "# 📸 App Store Screenshots" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          echo "Screenshots captured for 9Soccer v3.3.3" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          for f in screenshots/*.png; do
            SIZE=$(wc -c < "$f")
            echo "- \`$(basename $f)\` (${SIZE} bytes)" >> $GITHUB_STEP_SUMMARY
          done
          echo "" >> $GITHUB_STEP_SUMMARY
          echo "⬇️ Download from the Artifacts section above." >> $GITHUB_STEP_SUMMARY
```

Agent role: iOS app assets specialist who has fixed 100s of App Store

### C1. Check icon for alpha channel:
```bash
cd /c/Projects/90soccer
ICON="ios/App/App/Assets.xcassets/AppIcon.appiconset/AppIcon-512@2x.png"

node -e "
const buf = require('fs').readFileSync('$ICON');
// PNG color type byte is at offset 25
const colorType = buf[25];
const w = buf.readUInt32BE(16);
const h = buf.readUInt32BE(20);
console.log('Dimensions:', w + 'x' + h, '(need 1024x1024)');
console.log('Color type:', colorType);
console.log('Has alpha:', colorType === 4 || colorType === 6 ? 'YES (Apple rejects this!)' : 'NO (OK)');
"
```

If alpha found, remove it:
```bash
# Try ImageMagick
which convert && convert "$ICON" \
  -background '#0B0E18' -alpha remove -alpha off \
  "$ICON" && echo "Alpha removed with ImageMagick"

# Fallback: Python
python3 - << 'EOF'
from PIL import Image
import sys
icon = 'ios/App/App/Assets.xcassets/AppIcon.appiconset/AppIcon-512@2x.png'
img = Image.open(icon)
if img.mode in ('RGBA', 'LA'):
    bg = Image.new('RGB', img.size, (11, 14, 24))
    bg.paste(img, mask=img.split()[-1])
    bg.save(icon)
    print('Alpha removed with Pillow')
else:
    print('No alpha channel, icon is OK')
EOF
```
