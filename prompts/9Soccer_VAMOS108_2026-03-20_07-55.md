```typescript
// In BugReportContent.tsx
async function startRecording() {
  let stream: MediaStream | null = null;

  // Try screen capture first (works on desktop)
  try {
    stream = await navigator.mediaDevices.getDisplayMedia({
      video: { frameRate: 15 },
      audio: true
    });
    setRecordingType('screen');
  } catch {
    // Fallback: camera + mic (works on iOS)
    try {
      stream = await navigator.mediaDevices.getUserMedia({
        video: { facingMode: 'user' },
        audio: true
      });
      setRecordingType('camera');
    } catch {
      // Final fallback: audio only
      try {
        stream = await navigator.mediaDevices.getUserMedia({
          video: false,
          audio: true
        });
        setRecordingType('audio');
      } catch (err) {
        setError('Recording not supported on this device');
        return;
      }
    }
  }
  // continue with MediaRecorder...
}
```
