```typescript
// src/lib/levels.ts
export const LEVEL_XP = [0, 100, 250, 500, 900, 1400, 2000, 2800, 3800, 5000]
// Level 1 = 0 XP, Level 2 = 100 XP, etc.

export function getLevel(xp: number): number {
  return LEVEL_XP.filter(threshold => xp >= threshold).length
}

export function didLevelUp(oldXP: number, newXP: number): boolean {
  return getLevel(newXP) > getLevel(oldXP)
}

export function getLevelUnlock(level: number): string | null {
  const unlocks: Record<number, string> = {
    3: 'Tournament Mode ⚔️',
    5: 'CL Qualifier 🏅',
    7: 'Wager Mode 💰',
    10: 'Elite Badge 👑',
  }
  return unlocks[level] ?? null
}
```
